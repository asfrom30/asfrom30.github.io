# Vuejs

```js
// when to use :method
function onDeleteAssetBtn(clientAsset) {
  return function() {
    return clientAsset
      .removeFromServer()
      .then(() => {
        refreshTableData();
      })
      .catch(err => {
        console.error("can not delete");
        console.error(err);
      });
  };
}

// when use @click
function onDeleteAssetBtn(clientAsset) {
  clientAsset
    .removeFromServer()
    .then(() => {
      refreshTableData();
    })
    .catch(err => {
      console.error("can not delete");
      console.error(err);
    });
}
```
